USE bigfoot;
DROP TABLE IF EXISTS BFItem;
DROP TABLE IF EXISTS BFAccount;

CREATE TABLE BFAccount(
account_id INT AUTO_INCREMENT,
PRIMARY KEY(account_id),
BFname VARCHAR(255),
streetName VARCHAR(255),
streetNumber int,
city VARCHAR(255),
province VARCHAR(255),
postalCode VARCHAR(255));

CREATE TABLE BFItem(
barcodeNumber BIGINT,
itemName VARCHAR(255),
pictureAddress VARCHAR(255),
recyclingType ENUM('plastic', 'cardboard','garbage'),
timeAdded TIMESTAMP,
addedByID INT,
FOREIGN KEY(addedByID) 
REFERENCES BFAccount(account_id));
